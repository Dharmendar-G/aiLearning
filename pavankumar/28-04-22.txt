create table ZZOP0O nologging as
select	/*+ PARALLEL(32)*/ pa01.PARENT_ACCOUNT_ID  PARENT_ACCOUNT_ID,

pa01.STD_OR_CONTINGENCY  STD_OR_CONTINGENCY,

pa01.PCLIL1  WJXBFS1
from	ZZSP02	pa01
where	pa01.GODWFLAG17_1 = 1 



create table ZZMD0P nologging as
select	/*+ PARALLEL(32)*/ coalesce(pa11.STD_OR_CONTINGENCY, pa12.STD_OR_CONTINGENCY)  STD_OR_CONTINGENCY,

coalesce(pa11.PARENT_ACCOUNT_ID, pa12.PARENT_ACCOUNT_ID)  PARENT_ACCOUNT_ID,

pa11.WJXBFS1  PCLWP1,

pa11.WJXBFS2  PCLEP1,

pa12.WJXBFS1  PCLIL1
from	ZZOP0N	pa11

full outer join	ZZOP0O

  on 	(pa11.PARENT_ACCOUNT_ID = pa12.PARENT_ACCOUNT_ID and 

pa11.STD_OR_CONTINGENCY = pa12.STD_OR_CONTINGENCY)
